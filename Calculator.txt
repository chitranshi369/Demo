n1 = float(input("Enter the first number>>>>"))
n2 = float(input("Enter the second number>>>"))

#addition

print("{} + {} = ".format(n1, n2))
print(n1 + n2)

#subtraction

print("{} - {} = ".format(n1 , n2))
print(n1 - n2)

#multiplication

print("{} * {} = ".format(n1, n2))
print(n1 * n2)

#division

print("{} / {} = ".format(n1 , n2))
print(n1 / n2)
